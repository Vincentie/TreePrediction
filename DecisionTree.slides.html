#Data processing part. This part is to pick integrated factor like max_of_30day for future use.
pd.options.mode.chained_assignment = None

con = sqlite3.connect('Data/FAvgData.sqlite')
sql = "SELECT * from SHFEAvg"
data = pd.read_sql(sql, con, index_col = 'date')
data['close_after5day'] = None
data['max_of_30day'] = None
data['min_of_30day'] = None

for i in range(len(data)-30):
    data['close_after5day'][i] = data['Close'][i+5]
    #close_after5day = Close_(i+5)
    data['max_of_30day'][i+30] = max(data['High'][i:i+30])
    #max_of_30day = max{High_i, High_(i-1), ..., High_(i-29)}
    data['min_of_30day'][i+30] = min(data['Low'][i:i+30])
    #min_of_30day = min{Low_i, Low_(i-1), ..., Low_(i-29)}

data = data.dropna()
lables_raw = data['close_after5day']                #Data to be predicted.
lables = data['close_after5day'] > data['Close']    #To classify，we have to see if clsoe after 5 day is higher then today's close.

features = data.drop(['close_after5day'], axis = 1) 
features = features.drop(['High'], axis = 1)        #Drop High since we have max_of30day
features = features.drop(['Low'], axis = 1)         #Drop Low since we have min_of_30day

#Raw Prediction Tree Prediction
from sklearn.model_selection import train_test_split
from sklearn import preprocessing

#Standardize data.
scaler = preprocessing.StandardScaler().fit(features)
features_scaler = scaler.transform(features)

X_train,X_test, y_train, y_test = train_test_split(features_scaler, lables, test_size = 0.1, random_state = 0)

clf_tree = DecisionTreeClassifier(random_state=0)
clf_tree.fit(X_train, y_train)
print ("预测准确率为：%0.2f" % (clf_tree.score(X_test, y_test)))
